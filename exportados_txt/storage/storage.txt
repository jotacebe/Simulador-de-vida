"""
Ruta: storage/logs/event_logger.py
Responsabilidad: Suscriptor del EventBus encargado de centralizar la auditoría 
                 demográfica de la simulación mediante el sistema de logging.
"""
import logging
from typing import Dict, Any, Type

from events.population.person_born import PersonBornEvent
from events.population.person_died import PersonDiedEvent
from events.population.relationship_created import RelationshipCreatedEvent
from events.population.marriage_created import MarriageCreatedEvent
from events.population.divorce_occurred import DivorceOccurredEvent
from events.population.adoption_completed import AdoptionCompletedEvent
from events.event_bus import EventBus

# Configuración del logger
logger = logging.getLogger("DemographicsLog")

class EventLogger:
    """
    Sistema observador que traduce eventos del dominio en trazas audibles de ejecución.
    
    Centraliza todas las notificaciones demográficas, permitiendo que la lógica de
    negocio se mantenga pura y desacoplada de los formatos de salida (consola/archivo).
    """

    def __init__(self, event_bus: EventBus) -> None:
        """
        Suscribe el logger a todos los eventos demográficos relevantes del bus.
        """
        # Registro de suscripciones mediante un mapeo directo
        subscriptions: Dict[Type[Any], Any] = {
            PersonBornEvent: self.on_person_born,
            PersonDiedEvent: self.on_person_died,
            RelationshipCreatedEvent: self.on_relationship_created,
            MarriageCreatedEvent: self.on_marriage_created,
            DivorceOccurredEvent: self.on_divorce_occurred,
            AdoptionCompletedEvent: self.on_adoption_completed,
        }

        for event_type, handler in subscriptions.items():
            event_bus.subscribe(event_type, handler)

    def on_person_born(self, event: PersonBornEvent) -> None:
        logger.info(
            f"[Tick {event.tick}] 👶 NACIMIENTO: Entidad {event.entity_id} ha nacido. "
            f"Padres: {event.mother_id} y {event.father_id} | Género: {event.gender}"
        )

    def on_person_died(self, event: PersonDiedEvent) -> None:
        logger.info(
            f"[Tick {event.tick}] 🪦 DEFUNCIÓN: Entidad {event.entity_id} ha fallecido a los "
            f"{event.age:.2f} años. Causa: {event.cause}."
        )

    def on_relationship_created(self, event: RelationshipCreatedEvent) -> None:
        logger.info(
            f"[Tick {event.tick}] 💘 NOVIAZGO: Entidades {event.person_a_id} y {event.person_b_id} "
            f"han iniciado una relación (Compatibilidad: {event.compatibility_score:.2f})."
        )

    def on_marriage_created(self, event: MarriageCreatedEvent) -> None:
        logger.info(
            f"[Tick {event.tick}] 💍 MATRIMONIO: Entidades {event.person_a_id} y {event.person_b_id} "
            f"han consolidado su unión."
        )

    def on_divorce_occurred(self, event: DivorceOccurredEvent) -> None:
        logger.warning(
            f"[Tick {event.tick}] 💔 DIVORCIO: Entidades {event.person_a_id} y {event.person_b_id} "
            f"se han separado. Motivo: {event.reason}"
        )

    def on_adoption_completed(self, event: AdoptionCompletedEvent) -> None:
        parent_b_info: str = f" y {event.parent_b_id}" if event.parent_b_id else " (Soltero/a)"
        logger.info(
            f"[Tick {event.tick}] 👨‍👩‍👦 ADOPCIÓN: El menor {event.child_id} ha sido adoptado "
            f"por la entidad {event.parent_a_id}{parent_b_info}."
        )
